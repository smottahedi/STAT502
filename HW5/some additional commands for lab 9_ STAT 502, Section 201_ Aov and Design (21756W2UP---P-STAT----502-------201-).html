    some additional commands for lab 9

    data = read.table('apex.txt',header=T,sep='\t')
    rating = data[,1]
    A = as.factor(data[,2])
    fit = lm(rating~A)
    anova(fit)
    ybar = mean(rating)
    ls(anova(fit))
    mstr = anova(fit)$"Mean Sq"[1]
    mse = anova(fit)$"Mean Sq"[2]
    sig2mu = (mstr - mse) / 4

    library(lme4)
    fit2 = lmer(rating~1+(1|A))
    anova(fit2) # shows only fixed effects
    summary(fit2)
    ls(summary(fit2)) # not much here
    confint(fit2) # uses asymptotic normality
